PVar 함수를 후킹하는 방법으로, 수정된 데이터만 모아서 저장하도록 하는 PAWN · SQL 소스 예제입니다.

앞서 작성한 PAWN의 후킹 기법을 참고하시기 바랍니다.

  1. CREATE TABLE IF NOT EXISTS `playerdata` (
  2. `ID` INT(5) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  3. `Name` VARCHAR(32) NOT NULL DEFAULT ‘ ‘,
  4. `Varname` VARCHAR(64) NOT NULL DEFAULT ‘ ‘,
  5. `Vartype` INT(1) NOT NULL DEFAULT ‘0’,
  6. `Value` VARCHAR(512) NOT NULL DEFAULT ‘ ‘)

예제에 사용될 playerdata 테이블 생성문입니다.

각각 Name(플레이어이름), Varname(변수명), Vartype(변수타입), Value(값)입니다.

수정된 데이터만 저장하도록 하기 위해 SetPVar 함수에 앞서 SetPlayerData가 호출되도록 할 것입니다.

SetPlayerData는 변수 이름의 맨 앞에 p가 붙고, 값이 변경되었을 경우에 PlayerData 변수를 통해 이 변수를 저장할 것임을 기억하도록 할 것입니다.

NumSaveDatas는 불필요한 반복문을 줄이기 위해 PlayerData에서 사용되는 최대의 인덱스를 갖습니다.

아래는 사용된 함수들입니다.

  1. enum ePlayerData
  2. {
  3.         pdVarname[64],
  4.         pdVartype,
  5.         bool:pdSave
  6. }
  7. new     PlayerData[MAX_PLAYERS][MAX_PLAYER_DATAS][ePlayerData],
  8.         NumSaveDatas[MAX_PLAYERS];
  9.  
  10. stock SavePlayerData(playerid)
  11. {
  12.         new count = GetTickCount();
  13.         new str[3840];
  14.         if (IsPlayerNPC(playerid)) return 1;
  15.        
  16.         for (new i = 0; i < NumSaveDatas[playerid]; i++)
  17.                 if (strlen(PlayerData[playerid][i][pdVarname]) && PlayerData[playerid][i][pdSave])
  18.                 {
  19.                         new varname[64];
  20.                         strcpy(varname, PlayerData[playerid][i][pdVarname]);
  21.  
  22.                         format(str, sizeof(str), “UPDATE playerdata SET”);
  23.                         format(str, sizeof(str), “%s Vartype=%d”, str, GetPVarType(playerid, varname));
  24.                         switch (GetPVarType(playerid, varname))
  25.                         {
  26.                                 case PLAYER_VARTYPE_INT:        format(str, sizeof(str), “%s,Value=%d”, str, GetPVarInt(playerid, varname));
  27.                                 case PLAYER_VARTYPE_STRING: format(str, sizeof(str), “%s,Value=’%s'”, str, escape(GetPVarString(playerid, varname)));
  28.                                 case PLAYER_VARTYPE_FLOAT:  format(str, sizeof(str), “%s,Value=%f”, str,GetPVarFloat(playerid, varname));
  29.                         }
  30.                         format(str, sizeof(str), “%s WHERE Name=’%s’ And Varname=’%s'”, str, GetPlayerNameA(playerid), varname);
  31.                         mysql_query(str);
  32.                         PlayerData[playerid][i][pdSave] = false;
  33.                 }
  34.         NumSaveDatas[playerid] = 0;
  35.         printf(“SavePlayerData(%s): %dms”, GetPlayerNameA(playerid), GetTickCount() – count);
  36.         return 1;
  37. }
  38.  
  39. stock LoadPlayerData(playerid)
  40. {
  41.         new count = GetTickCount();
  42.         new     str[1024],
  43.                 receive[5][512],
  44.                 idx;
  45.         if (IsPlayerNPC(playerid)) return 1;
  46.        
  47.         format(str, sizeof(str), “SELECT * FROM playerdata WHERE Name=’%s'”, GetPlayerNameA(playerid));
  48.         mysql_query(str);
  49.         mysql_store_result();
  50.         for (new i = 0, t = mysql_num_rows(); i < t; i++)
  51.         {
  52.                 mysql_fetch_row(str, “|”);
  53.                 split(str, receive, ‘|’);
  54.                 idx = 2;
  55.  
  56.                 strcpy(PlayerData[playerid][i][pdVarname], receive[idx++]);
  57.                 PlayerData[playerid][i][pdVartype] = strval(receive[idx++]);
  58.                 switch (PlayerData[playerid][i][pdVartype])
  59.                 {
  60.                         case PLAYER_VARTYPE_INT:        SetPVarInt(playerid, PlayerData[playerid][i][pdVarname], strval(receive[idx++]));
  61.                         case PLAYER_VARTYPE_STRING: SetPVarString(playerid, PlayerData[playerid][i][pdVarname], receive[idx++]);
  62.                         case PLAYER_VARTYPE_FLOAT:  SetPVarFloat(playerid, PlayerData[playerid][i][pdVarname], floatstr(receive[idx++]));
  63.                 }
  64.                 PlayerData[playerid][i][pdSave] = false;
  65.         }
  66.         printf(“LoadPlayerData(%s): %dms”, GetPlayerNameA(playerid), GetTickCount() – count);
  67.         return 1;
  68. }
  69.  
  70. stock SetPlayerData(playerid, varname[], vartype, int_value, Float:float_value, string_value[])
  71. {
  72.         if (varname[0] != ‘p’) return 1;
  73.         new breaks;
  74.         switch (vartype)
  75.         {
  76.                 case PLAYER_VARTYPE_INT:        breaks = (int_value == GetPVarInt(playerid, varname)) ?true : false;
  77.                 case PLAYER_VARTYPE_FLOAT:      breaks = (float_value == GetPVarFloat(playerid, varname)) ? true : false;
  78.                 case PLAYER_VARTYPE_STRING:     breaks = (udb_hash(string_value) ==udb_hash(GetPVarString(playerid, varname))) ? true : false;
  79.         }
  80.         if (breaks) return 1;
  81.         new j = MAX_PLAYER_DATAS;
  82.         for (new i = 0; i < MAX_PLAYER_DATAS; i++)
  83.         {
  84.                 if (strlen(PlayerData[playerid][i][pdVarname])
  85.                 && !strcmp(PlayerData[playerid][i][pdVarname], varname, true))
  86.                 {
  87.                         PlayerData[playerid][i][pdSave] = true;
  88.                         if (NumSaveDatas[playerid] < i)
  89.                                 NumSaveDatas[playerid] = i+1;
  90.                         return 1;
  91.                 }
  92.                 else if (!strlen(PlayerData[playerid][i][pdVarname]) && j > i) j = i;
  93.         }
  94.         if (j == MAX_PLAYER_DATAS)
  95.         {
  96.                 printf(“%s님의 플레이어 데이터 수가 한도(%d)를 초과했습니다.”, GetPlayerNameA(playerid), MAX_PLAYER_DATAS);
  97.                 return 1;
  98.         }
  99.         InsertPlayerData(playerid, varname, vartype, int_value, float_value, string_value);
  100.         strcpy(PlayerData[playerid][j][pdVarname], varname);
  101.         PlayerData[playerid][j][pdVartype] = vartype;
  102.         PlayerData[playerid][j][pdSave] = false;
  103.         return 1;
  104. }
  105.  
  106. stock InsertPlayerData(playerid, varname[], vartype, int_value, Float:float_value, string_value[], encryption[]=“”)
  107. {
  108.         new str[1024], tmp[512];
  109.         format(str, sizeof(str), “INSERT INTO playerdata (Name,Varname,Vartype,Value)”);
  110.         format(str, sizeof(str), “%s VALUES (‘%s’,’%s’,%d,”, str, GetPlayerNameA(playerid), escape(varname), vartype);
  111.         switch (vartype)
  112.         {
  113.                 case PLAYER_VARTYPE_INT:        format(tmp, sizeof(tmp), “‘%d’)”, int_value);
  114.                 case PLAYER_VARTYPE_FLOAT:  format(tmp, sizeof(tmp), “‘%f’)”, float_value);
  115.                 case PLAYER_VARTYPE_STRING: format(tmp, sizeof(tmp), “‘%s’)”, string_value);
  116.         }
  117.         if (strlen(encryption))
  118.                 format(str, sizeof(str), “%s%s(%s)”, str, encryption, tmp);
  119.         else
  120.                 format(str, sizeof(str), “%s%s”, str, tmp);
  121.         mysql_query(str);
  122.         return 1;
  123. }
  124.  
  125. stock DeletePlayerData(playerid, varname[])
  126. {
  127.         if (varname[0] != ‘p’) return 1;
  128.         new str[256];
  129.         for (new i = 0; i < MAX_PLAYER_DATAS; i++)
  130.                 if (strlen(PlayerData[playerid][i][pdVarname])
  131.                 && !strcmp(PlayerData[playerid][i][pdVarname], varname, true))
  132.                 {
  133.                         format(str, sizeof(str), “DELETE FROM playerdata WHERE Name=’%s’ And Varname=’%s'”, GetPlayerNameA(playerid), varname);
  134.                         mysql_query(str);
  135.                         strcpy(PlayerData[playerid][i][pdVarname], chNullString);
  136.                         PlayerData[playerid][i][pdVartype] = PLAYER_VARTYPE_NONE;
  137.                         PlayerData[playerid][i][pdSave] = false;
  138.                         break;
  139.                 }
  140.         return 1;
  141. }

SetPVar 함수가 호출되기 전에 SetPlayerData > SetPVar가 호출되도록 후킹해 주셔야 하며, DeletePVar의 전에 DeletePlayerData가 호출되야 합니다.

InsertPlayerData 함수의 encryption 인자를 이용하면, 암호화 방식 또한 설정할 수 있습니다.

위 방법으로 실험해본 결과 저장 10ms, 불러오기 0ms의 속도가 나왔습니다만, 실전에서도 효율이 있을지는 잘 모르겠습니다.

시간이 부족해 설명보다는 소스 중심으로 올렸기 때문에 질문해 주신다면 기꺼이 답하겠습니다.
