function getAngle(x1,y1,x2,y2,i){
    var x1,y1,x2,y2,i;
    var dx = x2 – x1;
    var dy = y2 – y1;
   
   var rad= Math.atan2(dy, dx);
   var degree = (rad*180)/Math.PI ;
 
    //return degree,i;
    inputAngle(degree,i)
     
}
